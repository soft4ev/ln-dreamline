        document.getElementById('json-upload-input').onchange = (e) => {
            const file = e.target.files[0];
            if (!file) return;
            const reader = new FileReader();
            reader.onload = (re) => {
                try {
                    let data = JSON.parse(re.target.result);
                    if (!Array.isArray(data)) {
                        if (data.cards && Array.isArray(data.cards)) data = data.cards;
                        else if (data.mainCards && Array.isArray(data.mainCards)) data = data.mainCards;
                        else if (data.data && Array.isArray(data.data)) data = data.data;
                    }
                    if (Array.isArray(data)) {
                        showModal('导入字卡', `成功读取到 ${data.length} 条字卡。确定要导入并新增这些字卡吗？`, false, '', () => {
                            try {
                                // 修复：自动提取并兼容 JSON 中的分组及缺失的 id
                                data.forEach((c, idx) => {
                                    if (!c.id) c.id = 'tc_import_' + Date.now() + '_' + idx;
                                    if (!c.group) c.group = '未分组';
                                    if (c.enabled === undefined) c.enabled = true;
                                });
                                
                                // 合并字卡（将新字卡加在前面）
                                state.mainCards = [...data, ...state.mainCards];
                                
                                // 合并并去重分组
                                const newGroups = new Set(state.cardGroups);
                                data.forEach(c => newGroups.add(c.group));
                                state.cardGroups = Array.from(newGroups);
                                if (!state.cardGroups.includes('未分组')) {
                                    state.cardGroups.unshift('未分组');
                                }
                                
                                saveStateToStorage();
                                if (state.currentTab === 'custom') {
                                    renderCardGroupsAndList(); 
                                } else if (state.currentTab === 'emoji') {
                                    renderEmojiGrid();
                                }
                                closeModal();
                            } catch(e) {
                                console.error("导入执行出错", e);
                                closeModal(); // 防止卡死
                            }
                        });
                    } else {
                        showModal('导入失败', 'JSON格式不正确，必须是一个包含字卡的数组。请确保文件最外层是 [ ]。', false, '', () => closeModal());
                    }
                } catch (err) {
                    console.error("JSON解析失败", err);
                    showModal('解析失败', '文件不是规范的 JSON 格式，请检查标点符号是否正确。', false, '', () => closeModal());
                }
            };
            reader.readAsText(file);
            e.target.value = ''; 
        };

            clearTimeout(callTimeoutTimer);
            clearTimeout(simulatedActionTimer);
            clearInterval(callInterval);
            
            if (randomHangupTimer) {
                clearTimeout(randomHangupTimer);
                randomHangupTimer = null;
            }

            if (audioRingtone) {
                audioRingtone.pause();
                audioRingtone.currentTime = 0;
            }
            
            let msgText = '';
            
            if (reason === 'cancelled_by_initiator') {
                msgText = callInitiator === 'user' ? '□ᐊ　已取消' : '□ᐊ　对方已取消';
            } else if (reason === 'timeout') {
                msgText = callInitiator === 'user' ? '□ᐊ　对方未接听' : '□ᐊ　未接听';
            } else if (reason === 'rejected_by_receiver') {
                msgText = callInitiator === 'user' ? '□ᐊ　对方已拒绝' : '□ᐊ　已拒绝';
            } else if (reason === 'finished') {
                msgText = `□ᐊ　视频时长 ${formatDuration(callDuration)}`;
            }
            
            addMessage(msgText, callInitiator === 'user' ? 'right' : 'left');
            
            callState = 'idle';
            callWidget.classList.add('hidden');
            callWidget.classList.remove('mini');

            scheduleRandomCall();
        }

        function formatDuration(sec) {
            const h = Math.floor(sec / 3600);
            const m = Math.floor((sec % 3600) / 60);
            const s = sec % 60;
            if (h > 0) {
                return `${h.toString().padStart(2,'0')}:${m.toString().padStart(2,'0')}:${s.toString().padStart(2,'0')}`;
            }
            return `${m.toString().padStart(2,'0')}:${s.toString().padStart(2,'0')}`;
        }

        let isDragging = false;
        let hasMoved = false;
        let dragStartX, dragStartY, initialX, initialY;
        let dragRAF = null;

        function onDragStart(e) {
            if (e.target.closest('.call-btn') || e.target.closest('.call-minimize')) return;
            
            isDragging = true;
            hasMoved = false; 
            
            const clientX = e.type.includes('mouse') ? e.clientX : e.touches[0].clientX;
            const clientY = e.type.includes('mouse') ? e.clientY : e.touches[0].clientY;
            
            const rect = callWidget.getBoundingClientRect();
            dragStartX = clientX;
            dragStartY = clientY;
            initialX = rect.left;
            initialY = rect.top;
            
            callWidget.style.right = 'auto';
            callWidget.style.bottom = 'auto';
            callWidget.style.left = initialX + 'px';
            callWidget.style.top = initialY + 'px';

            callWidget.classList.add('dragging');

            document.addEventListener('mousemove', onDragMove, {passive: false});
            document.addEventListener('mouseup', onDragEnd);
            document.addEventListener('touchmove', onDragMove, {passive: false});
            document.addEventListener('touchend', onDragEnd);
        }

        function onDragMove(e) {
            if (!isDragging) return;
            
            const clientX = e.type.includes('mouse') ? e.clientX : e.touches[0].clientX;
            const clientY = e.type.includes('mouse') ? e.clientY : e.touches[0].clientY;
            
            let deltaX = clientX - dragStartX;
            let deltaY = clientY - dragStartY;

            if (Math.abs(deltaX) > 3 || Math.abs(deltaY) > 3) {
                hasMoved = true;
            }
            
            if (hasMoved) {
                e.preventDefault(); 
                
                if (dragRAF) cancelAnimationFrame(dragRAF);
                dragRAF = requestAnimationFrame(() => {
                    let newX = initialX + deltaX;
                    let newY = initialY + deltaY;
                    
                    const maxX = window.innerWidth - callWidget.offsetWidth;
                    const maxY = window.innerHeight - callWidget.offsetHeight;
                    newX = Math.max(0, Math.min(newX, maxX));
                    newY = Math.max(0, Math.min(newY, maxY));
                    
                    callWidget.style.left = newX + 'px';
                    callWidget.style.top = newY + 'px';
                });
            }
        }

        function onDragEnd(e) {
            if (isDragging && !hasMoved) {
                if (callWidget.classList.contains('mini')) {
                    toggleMiniCall();
                }
            }
            isDragging = false;
            hasMoved = false;
            callWidget.classList.remove('dragging');
            if (dragRAF) cancelAnimationFrame(dragRAF);

            document.removeEventListener('mousemove', onDragMove);
            document.removeEventListener('mouseup', onDragEnd);
            document.removeEventListener('touchmove', onDragMove);
            document.removeEventListener('touchend', onDragEnd);
        }

        callWidget.addEventListener('mousedown', onDragStart);
        callWidget.addEventListener('touchstart', onDragStart, {passive: false});

        async function requestReplySet(isAutoTrigger = false) {
            state.isSilenceMode = false;
            if (!isAutoTrigger && state.settings.ignore && Math.random() < 0.35) {
                state.isSilenceMode = true;
                return; 
            }

            const count = Math.floor(Math.random() * 3) + 1;
            // 核心逻辑：只从启用的（睁眼）字卡和Emoji中抽取
            const pool = [
                ...state.mainCards.filter(w => w.enabled).map(w => w.text),
                ...state.emojiState.filter(e => e.enabled).map(e => e.char)
            ];

            const newReplies = [];
            for (let i = 0; i < count; i++) {
                const text = pool.length > 0 ? pool[Math.floor(Math.random() * pool.length)] : "...";
                let quoteObj = null;
                const userMsgs = state.historyMessages.filter(m => m.side === 'right');
                if (state.settings.quote && userMsgs.length > 0 && Math.random() < 0.4) {
                    const src = userMsgs[Math.floor(Math.random() * userMsgs.length)];
                    quoteObj = { name: src.senderName, text: src.text.substring(0, 24) };
                }
                newReplies.push({ text, quote: quoteObj });
            }

            state.replyQueue.push(...newReplies);
            processReplyQueue();
        }

        async function processReplyQueue() {
            if (state.isProcessingQueue || state.replyQueue.length === 0) return;
            state.isProcessingQueue = true;
            while (state.replyQueue.length > 0) {
                const currentReply = state.replyQueue.shift();
                const waitTime = (Math.random() * (state.settings.maxSpeed - state.settings.minSpeed) + state.settings.minSpeed) * 1000;
                await new Promise(r => setTimeout(r, waitTime));
                let typingIndicator = null;
                if (state.settings.typing) {
                    typingIndicator = addMessage("", 'left', true);
                    await new Promise(r => setTimeout(r, Math.min(waitTime, 3000))); 
                    if (typingIndicator) typingIndicator.remove();
                }
                addMessage(currentReply.text, 'left', false, currentReply.quote);
                await new Promise(r => setTimeout(r, 600));
            }
            state.isProcessingQueue = false;
        }

        function checkOfflineActivity() {
            if (!state.settings.auto) return;
            const now = Date.now();
            const diffMinutes = (now - state.lastActivityTime) / (1000 * 60);

        // 渲染主列表与分组胶囊
        function renderCardGroupsAndList() {
            if (state.currentTab === 'emoji') {
                renderEmojiGrid();
                return;
            }

            const filterContainer = document.getElementById('group-filter-container');
            const selectContainer = document.getElementById('new-card-group-select');
            
            // 1. 统计各分组数量
            filterContainer.innerHTML = ''; selectContainer.innerHTML = '';
            const counts = { '全部': state.mainCards.length, '未分组': 0 };
            state.cardGroups.forEach(g => counts[g] = 0);
            state.mainCards.forEach(c => { 
                if(!c.group) c.group = '未分组';
                if(counts[c.group] !== undefined) counts[c.group]++; 
                else counts['未分组']++; 
            });

            // 2. 渲染顶部药丸筛选器
            filterContainer.innerHTML += `<div class="group-pill ${activeCardFilterGroup === '全部' ? 'active' : ''}" onclick="setCardFilter('全部')">全部 <span class="group-count">${counts['全部']}</span></div>`;
            filterContainer.innerHTML += `<div class="group-pill ${activeCardFilterGroup === '未分组' ? 'active' : ''}" onclick="setCardFilter('未分组')">未分组 <span class="group-count">${counts['未分组']}</span></div>`;
            
            state.cardGroups.forEach(group => {
                if(group !== '未分组' && !state.hiddenGroups.includes(group)) {
                    filterContainer.innerHTML += `<div class="group-pill ${activeCardFilterGroup === group ? 'active' : ''}" onclick="setCardFilter('${group}')">${group} <span class="group-count">${counts[group]}</span></div>`;
                }
            });

            // 3. 渲染底部新增下拉框
            selectContainer.innerHTML += `<option value="未分组">未分组</option>`;
            state.cardGroups.forEach(g => {
                if (g !== '未分组') selectContainer.innerHTML += `<option value="${g}">${g}</option>`;
            });
            // 智能选中当前查看的分组
            if (state.cardGroups.includes(activeCardFilterGroup)) {
                selectContainer.value = activeCardFilterGroup;
            }

            // 4. 数据过滤逻辑
            let displayIndices = [];
            
            // 预处理查重逻辑，找出重复文本
            let duplicateTexts = new Set();
            if (isDuplicateMode) {
                const textCounts = {};
                state.mainCards.forEach(c => textCounts[c.text] = (textCounts[c.text] || 0) + 1);
                for (let text in textCounts) if (textCounts[text] > 1) duplicateTexts.add(text);
            }

            const keyword = document.getElementById('card-search-input').value.trim().toLowerCase();

            state.mainCards.forEach((item, index) => {
                if (activeCardFilterGroup !== '全部' && item.group !== activeCardFilterGroup) return;
                if (keyword && !item.text.toLowerCase().includes(keyword)) return;
                if (isDuplicateMode && !duplicateTexts.has(item.text)) return;
                displayIndices.push(index);
            });

            // 5. 渲染字卡 DOM
            const listContainer = document.getElementById('card-list-container');
            listContainer.innerHTML = '';
            
            if (sortableInstance) {
                sortableInstance.destroy();
                sortableInstance = null;
            }

            if (displayIndices.length === 0) {
                listContainer.innerHTML = '';
                return;
            }

            displayIndices.forEach(originalIndex => {
                const card = state.mainCards[originalIndex];
                const eyeIcon = card.enabled ? 'fa-eye' : 'fa-eye-slash';
                const eyeColor = card.enabled ? 'active' : '';
                const hiddenClass = card.enabled ? '' : 'hidden-card';
                const isSelected = batchSelectedSet.has(originalIndex);
                const selectedClass = isSelected ? 'batch-selected' : '';
                
                const clickAction = isBatchMode ? `onclick="toggleBatchSelect(${originalIndex})"` : '';

                listContainer.innerHTML += `
                    <div class="text-card-item ${hiddenClass} ${selectedClass}" ${clickAction} data-index="${originalIndex}">
                        <div class="batch-checkbox"><i class="fa-solid fa-check" style="font-size: 12px; ${isSelected ? 'display:block' : 'display:none'}"></i></div>
                        <div class="text-card-content">${card.text}</div>
                        <div class="text-card-actions">
                            <div class="card-group-tag"><i class="fa-regular fa-folder" style="font-size:11px;"></i> ${card.group}</div>
                            <div class="card-action-icons">
                                <div class="card-icon-btn" onclick="editCard(${originalIndex}); event.stopPropagation();"><i class="fa-solid fa-pen-to-square"></i></div>
                                <div class="card-icon-btn" onclick="changeCardGroup(${originalIndex}); event.stopPropagation();"><i class="fa-solid fa-folder-open"></i></div>
                                <div class="card-icon-btn ${eyeColor}" onclick="toggleCardVis(${originalIndex}); event.stopPropagation();"><i class="fa-solid ${eyeIcon}"></i></div>
                                <div class="card-icon-btn delete" onclick="deleteCard(${originalIndex}); event.stopPropagation();"><i class="fa-regular fa-trash-can"></i></div>
                            </div>
                        </div>
                    </div>
                `;
            });

            // 仅在具体分组（非「全部」）且非特殊模式下允许拖拽排序
            if (!isBatchMode && !isSearchMode && !isDuplicateMode && activeCardFilterGroup !== '全部' && keyword === "") {
                sortableInstance = new Sortable(listContainer, {
                    animation: 150,
                    delay: 300, // 长按 0.3 秒触发拖拽
                    delayOnTouchOnly: true,
                    touchStartThreshold: 10, // 滑动超过 10px 取消长按判定，彻底修复手机端无法上下滚动的问题
                    forceFallback: true, // 强制启用 fallback 解决 iOS 无法跟随手指的视觉 BUG
                    fallbackClass: "sortable-fallback",
                    ghostClass: "sortable-ghost",
                    fallbackOnBody: true, // 将克隆元素挂载在 body 上，防止被父容器 overflow 切割
                    onEnd: function (evt) {
                        const oldIndex = evt.oldIndex;
                        const newIndex = evt.newIndex;
                        if (oldIndex === newIndex) return;
                        
                        // 核心：在原数组中，仅针对当前过滤出的位置进行元素对调，不影响其他分组的数据顺序
                        let slots = [...displayIndices];
                        let groupCards = slots.map(i => state.mainCards[i]);
                        
                        const [movedItem] = groupCards.splice(oldIndex, 1);
                        groupCards.splice(newIndex, 0, movedItem);
                        
                        slots.forEach((originalIdx, i) => {
                            state.mainCards[originalIdx] = groupCards[i];
                        });
                        
                        saveStateToStorage();
                        renderCardGroupsAndList();
                    }
                });
            }
        }

        function renameGroup(oldName) {
            const newName = prompt(`重命名分组【${oldName}】：`, oldName);
            if(newName && newName.trim() && newName.trim() !== oldName && !state.cardGroups.includes(newName.trim())) {
                const finalName = newName.trim(); 
                state.cardGroups[state.cardGroups.indexOf(oldName)] = finalName;
                if(state.hiddenGroups.includes(oldName)) { 
                    state.hiddenGroups[state.hiddenGroups.indexOf(oldName)] = finalName; 
                }
                state.mainCards.forEach(c => { if(c.group === oldName) c.group = finalName; }); 
                if (activeCardFilterGroup === oldName) activeCardFilterGroup = finalName;
                saveStateToStorage(); 
                openGroupManagerModal();
            }
        }

        function triggerAvatarUpload(target) { 
            state.activeTarget = target;

        document.getElementById('file-input').onchange = (e) => {
            const file = e.target.files[0];
            if (!file) return;
            const reader = new FileReader();
            reader.onload = (re) => {
                const img = new Image();
                img.onload = () => {
                    const canvas = document.createElement('canvas');
                    let w = img.width;
                    let h = img.height;
                    const maxDim = 800; 
                    if (w > maxDim || h > maxDim) {
                        if (w > h) { h = Math.round((h * maxDim) / w); w = maxDim; }
                        else { w = Math.round((w * maxDim) / h); h = maxDim; }
                    }
                    canvas.width = w;
                    canvas.height = h;
                    canvas.getContext('2d').drawImage(img, 0, 0, w, h);
                    const base64 = canvas.toDataURL('image/jpeg', 0.8);
                    
                    if (state.activeTarget === 'me') {
                        state.currentUser.avatar = base64;
                    } else {
                        state.friend.avatar = base64;
                    }
                    
                    saveStateToStorage();
                    refreshAllAvatars();
                };
                img.src = re.target.result;
            };
            reader.readAsDataURL(file);
            e.target.value = ''; 
        };

            const overlay = document.getElementById('modal-overlay');
            const confirmBtn = document.getElementById('modal-confirm-btn');
            if (type === 'clear') {
                document.getElementById('modal-title').innerText = '清空聊天记录';
                document.getElementById('modal-text').innerText = '确定删除所有聊天记录吗？';
                confirmBtn.onclick = () => {
                    const chatFlow = document.getElementById('chat-flow');
                    if(chatFlow) chatFlow.innerHTML = '';
                    state.historyMessages = [];
                    saveStateToStorage();
                    closeModal();
                    goBack();
                };
            } else {
                document.getElementById('modal-title').innerText = '恢复默认设置';
                document.getElementById('modal-text').innerText = '确定清除本地缓存并恢复默认吗？';
                confirmBtn.onclick = async () => { await localforage.clear(); location.reload(); };
            }
            overlay.classList.add('active');
        }

        window.onload = async () => {
            await loadStateFromStorage();
            syncSettingsUI();
            
            if (state.settings.push) {
                if (typeof requestNotificationPermission === "function") {
                    requestNotificationPermission();
                }
            }
            
            if (state.historyMessages.length > 0) {
                const msgs = [...state.historyMessages];
                state.historyMessages = [];
                msgs.forEach(m => addMessage(m.text, m.side, false, m.quote, m.time));
            }
            
            if (typeof checkOfflineActivity === "function") {
                checkOfflineActivity();
            }
            
            applyChatBackground();
            scheduleRandomCall();

            setInterval(() => {
                if (!state.settings.auto) return;
                const now = Date.now();
                const diffMinutes = (now - state.lastActivityTime) / (1000 * 60);
                if (diffMinutes >= state.settings.autoInterval) {
                    state.lastActivityTime = now;
                    requestReplySet(true);
                }
            }, 1000);
        };
    </script>
