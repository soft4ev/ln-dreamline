        <!-- 核心回复逻辑判定 -->
        async function requestReplySet(isAutoTrigger = false) {
            state.isSilenceMode = false;
            // 已读不回判定
            if (!isAutoTrigger && state.settings.ignore && Math.random() < 0.35) {
                state.isSilenceMode = true;
                return; 
            }

            const count = Math.floor(Math.random() * 3) + 1;
            const pool = [
                ...state.mainCards.filter(w => w.enabled).map(w => w.text),
                ...state.emojiState.filter(e => e.enabled).map(e => e.char)
            ];

            const newReplies = [];
            for (let i = 0; i < count; i++) {
                const text = pool.length > 0 ? pool[Math.floor(Math.random() * pool.length)] : "...";
                let quoteObj = null;
                // 引用过滤：只能引用 User 的消息
                const userMsgs = state.historyMessages.filter(m => m.side === 'right');
                if (state.settings.quote && userMsgs.length > 0 && Math.random() < 0.4) {
                    const src = userMsgs[Math.floor(Math.random() * userMsgs.length)];
                    quoteObj = { name: src.senderName, text: src.text.substring(0, 24) };
                }
                newReplies.push({ text, quote: quoteObj });
            }

            state.replyQueue.push(...newReplies);
            processReplyQueue();
        }
        // 串行执行引擎：严格控制等待 -> 正在输入 -> 消失 -> 发送
        async function processReplyQueue() {
            if (state.isProcessingQueue || state.replyQueue.length === 0) return;
            state.isProcessingQueue = true;
            // 动态生成等待间隔
            while (state.replyQueue.length > 0) {
                const currentReply = state.replyQueue.shift();
                const waitTime = (Math.random() * (state.settings.maxSpeed - state.settings.minSpeed) + state.settings.minSpeed) * 1000;
                await new Promise(r => setTimeout(r, waitTime));
                // 正在输入动画逻辑
                let typingIndicator = null;
                if (state.settings.typing) {
                    typingIndicator = addMessage("", 'left', true);
                    // 正在输入持续时间
                    await new Promise(r => setTimeout(r, Math.min(waitTime, 3000))); 
                    if (typingIndicator) typingIndicator.remove();
                }
                // 输出正式消息
                addMessage(currentReply.text, 'left', false, currentReply.quote);
                // 短暂的微小停顿，增加真实感
                await new Promise(r => setTimeout(r, 600));
            }
            state.isProcessingQueue = false;
        }

            // 如果离线时长超过了设置的自动回复间隔 (autoInterval)

                console.log(`离线补偿触发：离开${diffMinutes.toFixed(1)}分钟后重连`);

            }
        }

                    animation: 150,
                    // --- 以下是新增的逻辑代码 ---
                    delay: 200,            // 按住 200 毫秒（0.2秒）才开启拖拽
                    delayOnTouchOnly: true, // 关键：只在手机触摸屏上延迟，电脑鼠标点击依然秒触发
                    touchStartThreshold: 5, // 容错值：手指如果滑动超过 5 像素，就认为是想翻页，自动取消排序触发
                    // -------------------------
                    ghostClass: 'sortable-ghost',
                    chosenClass: 'sortable-chosen',
                    onEnd: function (evt) {
                        const oldIndex = evt.oldIndex;
                        const newIndex = evt.newIndex;
                        if (oldIndex === newIndex) return;
                        const movedItem = state.mainCards.splice(oldIndex, 1)[0];
                        state.mainCards.splice(newIndex, 0, movedItem);
                        saveStateToStorage();
                        renderList();

            resetDestiny();
        }

        function generateDestiny() {
            if(state.isGeneratingDestiny) return;
            state.isGeneratingDestiny = true;
            const startBtn = document.getElementById('btn-start-destiny');
            startBtn.disabled = true; startBtn.innerText = "命运正在指引中";
            document.getElementById('destiny-placeholder-text').style.display = "none";
            document.getElementById('destiny-status-area').style.display = "flex";
            document.getElementById('destiny-loading').style.display = "flex";
            document.getElementById('destiny-result-value').style.display = "none";
            setTimeout(() => {
                let res = "";
                if (state.destinyTab === 0) res = Math.random() < 0.5 ? "【正面】是" : "【反面】否";
                else if (state.destinyTab === 1) {
                    const r = Math.random();
                    if (r < 0.1) res = "【立筊】神迹"; else if (r < 0.4) res = "【圣筊】同意"; else if (r < 0.7) res = "【笑筊】不解"; else res = "【阴筊】否定";
                } else res = ["大吉", "吉", "小吉", "凶", "大凶"][Math.floor(Math.random() * 5)];
                state.lastDestinyResult = res;
                document.getElementById('destiny-loading').style.display = "none";
                const valEl = document.getElementById('destiny-result-value');
                valEl.innerText = res; valEl.style.display = "block";
                startBtn.style.display = "none";
                document.getElementById('destiny-action-row').style.display = "flex";
                state.isGeneratingDestiny = false;
            }, 1500);
        }

        function resetDestiny() {
            state.isGeneratingDestiny = false;
            document.getElementById('destiny-placeholder-text').style.display = "block";
            document.getElementById('destiny-status-area').style.display = "none";
            const btn = document.getElementById('btn-start-destiny');
            if (btn) { btn.style.display = "block"; btn.disabled = false; btn.innerText = "开始指引"; }
            const actionRow = document.getElementById('destiny-action-row');
            if (actionRow) actionRow.style.display = "none";
        }

        function sendDestinyResult() {
            const dir = document.getElementById('destiny-direction').value.trim();
            const type = ["硬币", "掷筊", "抽签"][state.destinyTab];
            const msg = dir ? `${dir}\n我使用了【${type}】：${state.lastDestinyResult}` : `我使用了【${type}】：${state.lastDestinyResult}`;
            addMessage(msg, 'right');
            goBack();
            requestReplySet();
        }

        const chatFlow = document.getElementById('chat-flow');
        const messageInput = document.getElementById('message-input');
        const sendBtn = document.getElementById('send-btn');

        messageInput.oninput = () => {
            const hasVal = messageInput.value.trim().length > 0;
            sendBtn.disabled = !hasVal;
            sendBtn.classList.toggle('active', hasVal);
        };
        // 核心修复：监听 Enter 键发送
        messageInput.onkeydown = (e) => {
            // 当按下 Enter 且没有按住 Shift 时触发发送
            if (e.key === 'Enter' && !e.shiftKey) {
                e.preventDefault(); // 阻止 textarea 默认的换行行为
                const text = messageInput.value.trim();
                if (text && !sendBtn.disabled) {
                    sendBtn.click(); // 模拟点击发送按钮
                }
            }
        };
        sendBtn.onclick = () => {
            const text = messageInput.value.trim();
            if(!text) return;
            addMessage(text, 'right', false, state.pendingQuote ? { ...state.pendingQuote } : null);
            clearPendingQuote();
            messageInput.value = '';
            sendBtn.disabled = true; sendBtn.classList.remove('active');
            requestReplySet();
        };

        function clearPendingQuote() {
            state.pendingQuote = null;
            document.getElementById('quote-preview').classList.remove('active');
        }

        function confirmAction(type) {
            const overlay = document.getElementById('modal-overlay');
            const confirmBtn = document.getElementById('modal-confirm-btn');
            if (type === 'clear') {
                document.getElementById('modal-title').innerText = '清空聊天记录';
                document.getElementById('modal-text').innerText = '确定删除所有聊天记录吗？';
                confirmBtn.onclick = () => { chatFlow.innerHTML = ''; state.historyMessages = []; closeModal(); goBack(); };
            } else {
                document.getElementById('modal-title').innerText = '恢复默认设置';
                document.getElementById('modal-text').innerText = '确定清除本地缓存并恢复默认吗？';
                confirmBtn.onclick = async () => { await localforage.clear(); location.reload(); };
            }
            overlay.classList.add('active');
        }

        function closeModal() { document.getElementById('modal-overlay').classList.remove('active'); }

        window.onload = async () => {
            await loadStateFromStorage();
            syncSettingsUI();
            // 根据设置检查通知权限
            if (state.settings.push) {
                if (typeof requestNotificationPermission === "function") {
                requestNotificationPermission();
            }
        }
            // 恢复历史聊天记录
            if (state.historyMessages.length > 0) {
                const msgs = [...state.historyMessages];
                state.historyMessages = [];
                msgs.forEach(m => addMessage(m.text, m.side, false, m.quote, m.time));
            }
            // 离线活动检查
            if (typeof checkOfflineActivity === "function") {
                checkOfflineActivity();
        }
            // 自动回复计时器
            setInterval(() => {
                if (!state.settings.auto) return;
                const now = Date.now();
                const diffMinutes = (now - state.lastActivityTime) / (1000 * 60);
                if (diffMinutes >= state.settings.autoInterval) {
                    state.lastActivityTime = now;
                    requestReplySet(true);
                }
            }, 1000);
        };
    </script>
