        async function saveStateToStorage() {
            try {

            try {
            const saved = await localforage.getItem(STORAGE_KEY);
            if (saved) {

        function navigateTo(pageId) {
            state.history.push(pageId);
            const el = document.getElementById(pageId);
            el.classList.add('page-active');
            if(pageId === 'page-settings' || pageId === 'page-comm-settings') syncSettingsUI();
            if(pageId === 'page-word-cards') renderList();
            if(pageId === 'page-destiny') resetDestiny();
        }

        function goBack() {
            if (state.history.length <= 1) return;
            const currentId = state.history.pop();
            const currentEl = document.getElementById(currentId);
            currentEl.classList.remove('page-active');
        }

        // --- 聊天执行引擎 ---
        function addMessage(text, side, isTyping = false, quote = null, oldTime = null) {
            const row = document.createElement('div');
            row.className = `msg-row msg-${side}`;
            const timeString = oldTime || new Date().toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });
            let quoteHtml = (quote && !isTyping) ? `<div class="quote-wrapper"><span class="quote-user">${quote.name}:</span>${quote.text}</div>` : '';
            const user = side === 'left' ? state.friend : state.currentUser;
            const displayText = isTyping ? '<div class="typing-dots"><span></span><span></span><span></span></div>' : text.replace(/\n/g, '<br>');

            row.innerHTML = `
                <div class="avatar"></div>
                <div class="bubble-container">
                    ${quoteHtml}
                    <div class="bubble">${displayText}</div>
                    ${!isTyping ? `<div class="timestamp">${timeString}</div>` : ''}
                </div>
            `;

            // --- 核心修复：添加头像点击功能 ---
            const avatarEl = row.querySelector('.avatar');
            avatarEl.onclick = () => {
                // 点击左侧头像（Dream）去设置，点击右侧头像（Me）也可以去设置
                navigateTo('page-settings'); 
            };
            
            // 引用功能逻辑优化
            if (!isTyping) {
                row.querySelector('.bubble').onclick = () => {
                    // 用户可以引用任何人的消息
                    state.pendingQuote = { name: user.name, text: text.substring(0, 30) };
                    document.getElementById('quote-preview-text').innerText = `${user.name}: ${text}`;
                    document.getElementById('quote-preview').classList.add('active');
                };
            }

            chatFlow.appendChild(row);
            renderAvatar(row.querySelector('.avatar'), user.avatar);
            setTimeout(() => row.scrollIntoView({ behavior: 'smooth', block: 'end' }), 50);
            
            if (!isTyping && !oldTime) {
                state.historyMessages.push({ text, senderName: user.name, side, quote, time: timeString });
                state.lastActivityTime = Date.now();// 只有真实发送消息才更新最后活动时间
                saveStateToStorage();
            }
            return row;
        }

        // 核心回复逻辑判定
        async function requestReplySet(isAutoTrigger = false) {
            // 打破沉默的情境：主动发送、主动回复触发
            state.isSilenceMode = false;

            // 沉默判定 (35% 概率)
            if (!isAutoTrigger && state.settings.ignore) {
                if (Math.random() < 0.35) {
                    state.isSilenceMode = true;
                    console.log("沉默判定：触发已读不回");
                    return; 
                }
            }

            // 决定回复条数 (1-3条)
            const count = Math.floor(Math.random() * 3) + 1;
            const pool = [
                ...state.customWords,
                ...state.presetWords.filter(w => w.enabled).map(w => w.text),
                ...state.emojiState.filter(e => e.enabled).map(e => e.char)
            ];

            const newReplies = [];
            for (let i = 0; i < count; i++) {
                const text = pool.length > 0 ? pool[Math.floor(Math.random() * pool.length)] : "...";
                let quoteObj = null;
                
                // 引用过滤：只能引用 User 的消息
                const userMsgs = state.historyMessages.filter(m => m.side === 'right');
                if (state.settings.quote && userMsgs.length > 0 && Math.random() < 0.4) {
                    const src = userMsgs[Math.floor(Math.random() * userMsgs.length)];
                    quoteObj = { name: src.senderName, text: src.text.substring(0, 24) };
                }
                newReplies.push({ text, quote: quoteObj });
            }

            state.replyQueue.push(...newReplies);
            processReplyQueue();
        }

        // 串行执行引擎：严格控制等待 -> 正在输入 -> 消失 -> 发送
        async function processReplyQueue() {
            if (state.isProcessingQueue || state.replyQueue.length === 0) return;
            state.isProcessingQueue = true;

            while (state.replyQueue.length > 0) {
                const currentReply = state.replyQueue.shift();
                
                // 1. 动态生成等待间隔
                const waitTime = (Math.random() * (state.settings.maxSpeed - state.settings.minSpeed) + state.settings.minSpeed) * 1000;
                await new Promise(r => setTimeout(r, waitTime));

                // 2. 正在输入动画逻辑
                let typingIndicator = null;
                if (state.settings.typing) {
                    typingIndicator = addMessage("", 'left', true);
                    // 正在输入持续时间
                    await new Promise(r => setTimeout(r, Math.min(waitTime, 3000))); 
                    if (typingIndicator) typingIndicator.remove();
                }

                // 3. 输出正式消息
                addMessage(currentReply.text, 'left', false, currentReply.quote);
                showNotification(state.friend.name, currentReply.text);
                
                // 短暂的微小停顿，增加真实感
                await new Promise(r => setTimeout(r, 600));
            }

            state.isProcessingQueue = false;
        }

        // --- 核心改动：离线补发机制 ---
        function checkOfflineActivity() {
            if (!state.settings.auto) return;
            const now = Date.now();
            const diffMinutes = (now - state.lastActivityTime) / (1000 * 60);

            // 如果离线时长超过了设置的自动回复间隔 (autoInterval)
            if (diffMinutes >= state.settings.autoInterval) {
                console.log(`离线补偿触发：离开${diffMinutes.toFixed(1)}分钟后重连`);
                state.lastActivityTime = now; // 立即重置，防止重复触发
                saveStateToStorage();
                
                // 延迟一小会儿触发回复，让用户进页面后能看到“正在输入”动画
                setTimeout(() => {
                    requestReplySet(true);
                }, 1200);
            }
        }

        // 主动回复计时检查 (每秒执行一次，不依赖页面状态)
        setInterval(() => {
            if (!state.settings.auto) return;
            const now = Date.now();
            const diffMinutes = (now - state.lastActivityTime) / (1000 * 60);
            if (diffMinutes >= state.settings.autoInterval) {
                console.log("主动回复触发：打破沉默");
                state.lastActivityTime = now; // 重置计时
                requestReplySet(true);
            }
        }, 1000);

        // 监听从后台切换回来的动作
        document.addEventListener('visibilitychange', () => {
            if (document.visibilityState === 'visible') {
                checkOfflineActivity();
            }
        });

        // --- UI 同步与列表渲染 ---
        function toggleSwitch(key) {
            state.settings[key] = !state.settings[key];
            if (key === 'push' && state.settings[key]) requestNotificationPermission();
            saveStateToStorage();
            syncSettingsUI();
        }

        function updateSlider(type, val) {
            const v = parseInt(val);
            if (type === 'min') state.settings.minSpeed = v;
            if (type === 'max') state.settings.maxSpeed = Math.max(v, state.settings.minSpeed + 1);
            if (type === 'auto') state.settings.autoInterval = v;
            saveStateToStorage();
            syncSettingsUI();
        }

            document.getElementById('val-min').innerText = `${state.settings.minSpeed}秒钟`;
            document.getElementById('range-min').value = state.settings.minSpeed;
            document.getElementById('val-max').innerText = `${state.settings.maxSpeed}秒钟`;
            document.getElementById('range-max').value = state.settings.maxSpeed;
            document.getElementById('val-auto').innerText = `${state.settings.autoInterval}分钟`;
            document.getElementById('range-auto').value = state.settings.autoInterval;
        }

        function switchTab(tab) {
            state.currentTab = tab;
            document.querySelectorAll('.tab-btn').forEach(b => b.classList.toggle('active', b.id === `tab-${tab}`));
            document.getElementById('add-word-bar').style.display = (tab === 'custom') ? 'flex' : 'none';
            renderList();
        }

        function renderList() {
            const container = document.getElementById('cards-list');
            container.innerHTML = '';
            if (state.currentTab === 'custom') {
                state.customWords.forEach((word, index) => {
                    const card = document.createElement('div');
                    card.className = 'card-item';
                    card.innerHTML = `<div class="card-text">${word}</div><i class="fa-solid fa-trash-can card-icon" onclick="deleteCustomWord(${index})"></i>`;
                    container.appendChild(card);
                });
            } else if (state.currentTab === 'preset') {
                state.presetWords.forEach((word, index) => {
                    const card = document.createElement('div');
                    card.className = 'card-item';
                    card.innerHTML = `<div class="card-text">${word.text}</div><i class="fa-solid ${word.enabled ? 'fa-eye active' : 'fa-eye-slash'} card-icon" onclick="togglePreset(${index})"></i>`;
                    container.appendChild(card);
                });
            } else if (state.currentTab === 'emoji') {
                const grid = document.createElement('div');
                grid.className = 'emoji-grid';
                state.emojiState.forEach((item, index) => {
                    const box = document.createElement('div');
                    box.className = `emoji-box ${item.enabled ? '' : 'disabled'}`;
                    box.innerHTML = `${item.char}<i class="fa-solid ${item.enabled ? 'fa-eye' : 'fa-eye-slash'} emoji-eye-status"></i>`;
                    box.onclick = () => toggleEmoji(index);
                    grid.appendChild(box);
                });
                container.appendChild(grid);
            }
        }

        function saveNewWord() {
            const input = document.getElementById('new-word-input');
            const val = input.value.trim();
            if (!val) return;
            state.customWords.unshift(val);
            input.value = '';
            checkInputReady();
            saveStateToStorage();
            renderList();
        }
        function checkInputReady() {
            const val = document.getElementById('new-word-input').value.trim();
            document.getElementById('btn-add').classList.toggle('ready', val.length > 0);
        }
        function deleteCustomWord(i) { state.customWords.splice(i, 1); saveStateToStorage(); renderList(); }
        function togglePreset(i) { state.presetWords[i].enabled = !state.presetWords[i].enabled; saveStateToStorage(); renderList(); }
        function toggleEmoji(i) { state.emojiState[i].enabled = !state.emojiState[i].enabled; saveStateToStorage(); renderList(); }

            state.activeTarget = target;
            const input = document.getElementById('nickname-edit-input');
            input.value = target === 'me' ? state.currentUser.name : state.friend.name;
            navigateTo('page-nickname-edit');
            input.oninput();

        function saveNickname() {
            const val = document.getElementById('nickname-edit-input').value.trim();
            if (state.activeTarget === 'me') state.currentUser.name = val;
            else { state.friend.name = val; document.getElementById('chat-title').innerText = val; }

            syncSettingsUI();

        function triggerAvatarUpload(target) { state.activeTarget = target; document.getElementById('file-input').click(); }
        document.getElementById('file-input').onchange = (e) => {
            const file = e.target.files[0];
            if (!file) return;
            const reader = new FileReader();
            reader.onload = (re) => {
                const img = new Image();
                img.onload = () => {
                    const canvas = document.createElement('canvas');
                    canvas.width = 120; canvas.height = 120;
                    canvas.getContext('2d').drawImage(img, 0, 0, 120, 120);
                    const base64 = canvas.toDataURL('image/jpeg', 0.8);
                    if (state.activeTarget === 'me') state.currentUser.avatar = base64;
                    else state.friend.avatar = base64;
                    saveStateToStorage();
                    refreshAllAvatars();
                };
                img.src = re.target.result;
            };
            reader.readAsDataURL(file);
        };
        function renderAvatar(el, src) {
            if (!el) return;
            if (src) { el.style.backgroundImage = `url(${src})`; el.innerHTML = ''; } 
            else { el.style.backgroundImage = 'none'; el.innerHTML = '<i class="fa-solid fa-user"></i>'; }
        }
        function refreshAllAvatars() {
            renderAvatar(document.getElementById('avatar-me'), state.currentUser.avatar);
            renderAvatar(document.getElementById('avatar-friend'), state.friend.avatar);
            document.querySelectorAll('.msg-left .avatar').forEach(el => renderAvatar(el, state.friend.avatar));
            document.querySelectorAll('.msg-right .avatar').forEach(el => renderAvatar(el, state.currentUser.avatar));
        }

            if(state.isGeneratingDestiny) return;

        function generateDestiny() {
            if(state.isGeneratingDestiny) return;
            state.isGeneratingDestiny = true;
            const startBtn = document.getElementById('btn-start-destiny');
            startBtn.disabled = true; startBtn.innerText = "命运正在指引中";
            document.getElementById('destiny-placeholder-text').style.display = "none";
            document.getElementById('destiny-status-area').style.display = "flex";
            document.getElementById('destiny-loading').style.display = "flex";
            document.getElementById('destiny-result-value').style.display = "none";
            setTimeout(() => {
                let res = "";
                if (state.destinyTab === 0) res = Math.random() < 0.5 ? "【正面】是" : "【反面】否";
                else if (state.destinyTab === 1) {
                    const r = Math.random();
                    if (r < 0.1) res = "【立筊】神迹"; else if (r < 0.4) res = "【圣筊】同意"; else if (r < 0.7) res = "【笑筊】不解"; else res = "【阴筊】否定";
                } else res = ["大吉", "吉", "小吉", "凶", "大凶"][Math.floor(Math.random() * 5)];
                state.lastDestinyResult = res;
                document.getElementById('destiny-loading').style.display = "none";
                const valEl = document.getElementById('destiny-result-value');
                valEl.innerText = res; valEl.style.display = "block";
                startBtn.style.display = "none";
                document.getElementById('destiny-action-row').style.display = "flex";
                state.isGeneratingDestiny = false;
            }, 1500);
        }
        function resetDestiny() {
            state.isGeneratingDestiny = false;
            document.getElementById('destiny-placeholder-text').style.display = "block";
            document.getElementById('destiny-status-area').style.display = "none";
            const btn = document.getElementById('btn-start-destiny');
            btn.style.display = "block"; btn.disabled = false; btn.innerText = "开始指引";
            document.getElementById('destiny-action-row').style.display = "none";
        }

    <script type="text/file" data-filename="sw.js">
    const CACHE_NAME = 'light-night-dreamline-v1';
    const ASSETS_TO_CACHE = [
      './index.html',
      'https://cdnjs.cloudflare.com/ajax/libs/font-awesome/6.4.2/css/all.min.css'
      'https://cdnjs.cloudflare.com/ajax/libs/localforage/1.10.0/localforage.min.js'
    ];

    // 安装阶段：缓存核心资源
    self.addEventListener('install', (event) => {
      event.waitUntil(
        caches.open(CACHE_NAME).then((cache) => {
          return cache.addAll(ASSETS_TO_CACHE);
        })
      );
    });

    // 激活阶段：清理旧缓存
    self.addEventListener('activate', (event) => {
      event.waitUntil(
        caches.keys().then((cacheNames) => {
          return Promise.all(
            cacheNames.map((name) => {
              if (name !== CACHE_NAME) return caches.delete(name);
            })
          );
        })
      );
    });

    // 拦截请求：优先从缓存读取
    self.addEventListener('fetch', (event) => {
      event.respondWith(
        caches.match(event.request).then((response) => {
          return response || fetch(event.request);
        })
      );
    });
    </script>
